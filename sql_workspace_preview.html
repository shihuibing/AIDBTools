2        SUM(oi.quantity) AS total_qty,
3        SUM(oi.quantity * oi.price) AS total_sales
4FROM orders o
5JOIN order_items oi ON o.id = oi.order_id
6JOIN products p ON oi.product_id = p.id
7JOIN categories c ON p.category_id = c.id
8WHERE o.status = 'completed'
9  AND o.created_at >= '2026-01-01'
10GROUP BY p.id, p.name, p.price, c.category_name
11ORDER BY total_sales DESC
12LIMIT 10;
